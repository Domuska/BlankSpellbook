spell
	ID (autogenerated)
		+ attackType
		+ attackRoll
		+ castingTime
		+ groupName
		+ hitDamageOrEffect
		+ missDamage
		+ name
		+ playerNotes
		+ rechargeTime
		spellListId
		+ target
		trigger
		+ adventurerFeat
		+ championFeat
		+ epicFeat


spell_group
    ID (daily spells/id)
        groupName1 : true (spell/id/groupName)
        groupName2 : true (spell/id/groupName)
        ...
        groupNameN : true (spell/id/groupName)
